#include<iostream>
#include<iomanip>

using namespace std;
void unos(int[][10], int);
void ispis(int[][10], int);

int main()
{
	int n, matrica[10][10];

	cout << "Izaberite velicinu matrice: ";
	do {
		cin >> n;
		if (n < 1 || n>10)
			cout << "Unos nije validan!" << endl;
	} while (n < 1 || n>10);

	unos(matrica, n);
	cout << endl << endl;

	cout << "Dobili ste sl matricu: " << endl;
	ispis(matrica, n);
	cout << endl;

	if (n % 2 == 0)
		cout << "------> 0" << endl << endl;
	else
		cout << "------> 1" << endl << endl;



}
void unos(int niz[][10], int n)
{
	for (int i = 0; i < n; i++)
		for (int j = 0; j < n; j++)
			niz[i][j] = rand() % 100;
}
void ispis(int niz[][10], int n)
{
	cout << endl;
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++)
			cout << setw(3) << niz[i][j] << " ";
		cout << endl;
	}
}
